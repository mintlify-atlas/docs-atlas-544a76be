## Create Your Project

Generate a new backend application using the CLI:

```bash
create-backend-app my-optics-api
```

This creates a complete MVC project structure:

```bash
✅ Proyecto my-optics-api creado con éxito
```

<Note>
  If you don't specify a project name, it defaults to `project-root`.
</Note>

## Project Structure

The CLI generates the following structure:

```
my-optics-api/
├── src/
│   ├── config/          # Configuration files
│   ├── controllers/     # Request handlers
│   ├── models/          # TypeORM entities
│   ├── routes/          # Express routes
│   ├── services/        # Business logic
│   ├── middlewares/     # Custom middleware
│   ├── validators/      # Input validation
│   ├── errors/          # Error handling
│   ├── utils/           # Helper functions
│   ├── app.ts           # Express app configuration
│   ├── server.js        # Server entry point
│   └── data-source.ts   # TypeORM configuration
├── tests/               # Test files
├── .env                 # Environment variables
├── .gitignore
├── package.json
└── README.md
```

## Install Dependencies

Navigate to your project and install dependencies:

<Steps>
  <Step title="Enter Project Directory">
    ```bash
    cd my-optics-api
    ```
  </Step>

  <Step title="Install Packages">
    ```bash
    npm install
    ```
    
    This installs the required dependencies:
    - `express` - Web framework
    - `typeorm` - ORM for database operations
    - `pg` - PostgreSQL driver
    - `bcrypt` - Password hashing
    - `jsonwebtoken` - JWT authentication
    - `express-rate-limit` - Rate limiting
    - `dotenv` - Environment variables
  </Step>
</Steps>

## Configure Database

Update the `.env` file with your PostgreSQL credentials:

```bash .env
PORT=3000

DB_HOST=localhost
DB_PORT=5432
DB_USER=postgres
DB_PASSWORD=your_password
DB_NAME=my_optics_db

NODE_ENV=development
```

<Warning>
  Make sure the database exists before starting the server. The TypeORM configuration uses `synchronize: true` in development, which automatically creates tables.
</Warning>

## Generate Your First Model

Use the model generator to create a TypeORM entity. For example, let's create a `Cliente` (Customer) model:

```bash
node ../generate-model.js
```

The generator creates entities based on the schema defined in `generate-model.js`. Here's an example of a generated model:

```typescript src/models/Cliente.ts
import { Column, Entity, JoinColumn, ManyToOne, OneToMany, PrimaryGeneratedColumn } from "typeorm";
import { Cita } from "./Cita";
import { Venta } from "./Venta";

@Entity("cliente")
export class Cliente {

  @PrimaryGeneratedColumn()
  id_cliente!: number;

  @Column()
  tipoDocumento!: string;

  @Column()
  documento!: string;

  @Column()
  nombre!: string;

  @Column({ unique: true })
  correo!: string;

  @Column({ nullable: true })
  telefono?: string;

  @Column({ type: "boolean" })
  estado!: boolean;

  @OneToMany(() => Cita, (x) => x.cliente)
  citas!: Cita[];

  @OneToMany(() => Venta, (x) => x.cliente)
  ventas!: Venta[];

}
```

```typescript src/controllers/ClienteController.ts
export const getAllCliente = async (_req: Request, res: Response): Promise<void> => {
  try {
    const clienteRepo = AppDataSource.getRepository(Cliente);
    const items = await clienteRepo.find();
    res.json({ success: true, data: items });
  } catch (error) {
    res.status(500).json({ success: false, message: "Error al obtener Cliente", error });
  }
};

export const createCliente = async (req: Request, res: Response): Promise<void> => {
  try {
    const clienteRepo = AppDataSource.getRepository(Cliente);
    const required = ["tipoDocumento", "documento", "nombre", "correo"];
    const missing = required.filter(k => req.body[k] === undefined);
    if (missing.length) { 
      res.status(400).json({ 
        success: false, 
        message: `Campos requeridos: ${missing.join(", ")}` 
      }); 
      return; 
    }
    const obj = clienteRepo.create();
    obj.tipoDocumento = req.body.tipoDocumento;
    obj.documento = req.body.documento;
    obj.nombre = req.body.nombre;
    obj.correo = req.body.correo;
    obj.telefono = req.body.telefono;
    obj.estado = req.body.estado !== undefined ? req.body.estado : true;

    await clienteRepo.save(obj);
    res.status(201).json({ 
      success: true, 
      message: "Cliente creado exitosamente", 
      data: obj 
    });
  } catch (error) {
    res.status(500).json({ 
      success: false, 
      message: "Error al crear Cliente", 
      error 
    });
  }
};
```

```typescript src/routes/ClienteRoutes.ts
import { Router } from "express";
import { 
  getAllCliente, 
  getClienteById, 
  createCliente, 
  updateCliente, 
  deleteCliente, 
  toggleEstadoCliente 
} from "../controllers/ClienteController";

const router = Router();

// GET    /           → listar todos
router.get("/", getAllCliente);

// GET    /:id        → obtener uno por ID
router.get("/:id", getClienteById);

// POST   /           → crear nuevo
router.post("/", createCliente);

// PUT    /:id        → actualizar
router.put("/:id", updateCliente);

// DELETE /:id        → eliminar (valida dependencias)
router.delete("/:id", deleteCliente);

// PATCH  /:id/estado → activar / inactivar
router.patch("/:id/estado", toggleEstadoCliente);

export { router as clienteRouter };
```

### Delete Customer

```bash
curl -X DELETE http://localhost:3000/api/clientes/1
```

<Warning>
  Deletion validates foreign key constraints. If the customer has related records (appointments, sales), the operation will fail with a 409 Conflict error.
</Warning>

## Available Endpoints

All generated models have the same CRUD endpoints:

| Method | Endpoint | Description |
|--------|----------|-------------|
| GET | `/api/<resource>` | List all records |
| GET | `/api/<resource>/:id` | Get one record by ID |
| POST | `/api/<resource>` | Create new record |
| PUT | `/api/<resource>/:id` | Update record |
| DELETE | `/api/<resource>/:id` | Delete record (validates dependencies) |
| PATCH | `/api/<resource>/:id/estado` | Toggle active/inactive status |
